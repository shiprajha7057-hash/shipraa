{
    #include
using namespace std;

int main() {

    cin >> n;

    int arr[n];

    cout << "Enter elements:\n";
    for(int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Frequency calculation
    for(int i = 0; i < n; i++) {
        int count = 1;

        // Skip if already counted
        if(arr[i] == -1)
            continue;

        for(int j = i + 1; j < n; j++) {
            if(arr[i] == arr[j]) {
                count++;
                arr[j] = -1; // mark as visited
            }
        }

    }

    return 0;
}

double squareRoot(double n) {
    double x = n;       
    double y = 1.0;    
    double e = 0.00001; 

    while (x - y > e || y - x > e) {
        x = (x + y) / 2;
        y = n / x;

    return x;
